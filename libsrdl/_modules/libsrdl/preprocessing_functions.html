def import_images_and_rt(images_path, rt_path, label):
    """
    Import images and RT Structures from file and convert them to nifti.

    :param images_path: path of the directory containing dicom images (i.e.: CT) (Path)
    :param rt_path: path of the dicom RT structure (Path)
    :param label: ROI name (i.e.: PTV/CTV) (str)
    :return: image: nifti image (i.e.: CT) (nifti)
    :return: mask_nifti: nifti mask of the ROI previously selected (nifti)
    """

    image = rts_to_nifti.read_dicom_image(images_path)
    mask_rt = rts_to_nifti.read_dicom_struct_file(rt_path)
    mask_nifti = rts_to_nifti.transform_point_set_from_dicom_struct(image, mask_rt)

    # consider also cases where the labels are not written always in the same way
    uniform_roi_names = [
        label["short_label"].upper() in roi_name.upper() for roi_name in mask_nifti[1]
    ]
    ind_roi_name = np.where(np.array(uniform_roi_names) == True)[0]
    if len(ind_roi_name) > 1:
        print(
            f"Allert!! Found more rois with {label} : {str(np.array(mask_nifti[1])[ind_roi_name].tolist())}"
        )
        uniform_roi_names = [
            label["full_label"].upper() in roi_name.upper() for roi_name in mask_nifti[1]
        ]
        ind_roi_name = np.where(np.array(uniform_roi_names) == True)[0]
        label_found = mask_nifti[1][ind_roi_name[0]]
        print(f"It was taken the roi: {label_found}")

    return image, mask_nifti[0][ind_roi_name[0]]



[docs]def create_resized_image(image, mask_nifti, isocentre, size):
    """
    Function to resize an image and its mask to a squared image.

    :param image: image to be rescaled (nifti)
    :param mask_nifti: mask of the chosen ROI (nifti)
    :param isocentre: isocentre of the image (class Isocentre)
    :param size: number of pixels of final image (int)
    :return: image_resized: image resized (array)
    :return: mask_3d_resized: array of mask resized (array)
    """

    x_left = int(isocentre.x - size / 2)
    x_right = int(isocentre.x + size / 2)
    y_up = int(isocentre.y - size / 2)
    y_down = int(isocentre.y + size / 2)

    # cropping
    image_resized = image[x_left:x_right, y_up:y_down]
    # resize the mask according to the size of the resized image
    mask_3d_resized = mask_nifti[x_left:x_right, y_up:y_down]
    # make it a "true mask" by converting it to an array
    # TODO we must convert the image to array even if we don't crop it
    image_resized = sitk.GetArrayFromImage(image_resized)
    mask_3d_resized = sitk.GetArrayFromImage(mask_3d_resized)
    return image_resized, mask_3d_resized




[docs]def rescale_color(image):
    """
    Function to rescale color of an image in range [0,1].

    :param image: image to rescale (nifti)
    :return: image: image resized (nifti)
    """

    image = sitk.GetArrayFromImage(image)
    image = (image - np.min(image)) / (np.max(image) - np.min(image))
    image = sitk.GetImageFromArray(image)
    return image

[docs]def check_xy_size(images, using_mask):
    """
    Check if size of images are the same.

    :param images: images to check (numpy array)
    :param using_mask: type of image to check (string)
    return: check:
    """

    if using_mask:
        imag_type = "mask"
    else:
        imag_type = "image"
    x_shape, y_shape = [], []
    for i in range(len(images)):
        x_shape.append(images[i].shape[1])
        y_shape.append(images[i].shape[2])
    check_x = len(np.unique(x_shape)) == 1
    check_y = len(np.unique(y_shape)) == 1
    check_xy = (np.unique(x_shape) == np.unique(y_shape))[0]
    if check_x != True or check_y != True or check_xy != True:
        print(
            f"Allert x, y dimensions for {imag_type} are not the same for all the images or images are not square. Aborting!"
        )
    check = check_x and check_y and check_xy

    assert check

[docs]def from_mask_to_bounding_box(mask, scaling):
    """
    Function to create class labels and bounding boxes starting from segmented ROIs.

    :param mask: segmented ROI of image (boolean matrix)
    :param scaling:
    :return: labels: boolean array (1:is present), shape: (n_images,1)
    :return: boxes: 2D array storing bounding boxes coordinates (x min, y min, x max, y max)
            normalized with image shape, shape:(n_images,4)
    """

    n_images, y_shape, x_shape = mask.shape[0], mask.shape[1], mask.shape[2]
    if scaling != True:
        y_shape, x_shape = 1, 1
    labels = np.zeros((n_images, 1))
    boxes = np.zeros((n_images, 4))
    # boxes = np.full([n_images, 4], np.nan)

    for i in range(n_images):
        if np.any(mask[i, :, :] != 0):
            labels[i] = 1
            index_roi_y, index_roi_x = np.where(mask[i, :, :] != 0)
            boxes[i] = [
                min(index_roi_x) / x_shape,
                min(index_roi_y) / y_shape,
                max(index_roi_x) / x_shape,
                max(index_roi_y) / y_shape,
            ]
    return labels, boxes

[docs]def crop_loaded_images(images, final_size, image_isocentre):
    """
    crop images to final_size from .nii file.

    :param images: CT images to resize (numpy array)
    :param final_size: final size (int)
    :param image_isocentre: pixel position of isocentre (int 2D array)
    :return: images: resized images (numpy array)
    """

    x_left = int(image_isocentre.x - final_size / 2)
    x_right = int(image_isocentre.x + final_size / 2)
    y_up = int(image_isocentre.y - final_size / 2)
    y_down = int(image_isocentre.y + final_size / 2)

    images = np.moveaxis(images, 0, -1)
    images = images[x_left:x_right, y_up:y_down]
    images = np.moveaxis(images, -1, 0)

    return images
