[docs]def plot_sample(images, classes, boxes, pclasses=None, pboxes=None, scaling: bool = True, plot_predictions: bool = False,
     pathout: Path = Path("."), save=True) -> None:
    """
    Print of some images with real bounding box and its prediction.

    :param images: images to plot (numpy array)
    :param classes: labels of images (boolean array)
    :param boxes: bounding box to plot (array)
    :param pclasses: int variable describing type of target (int)
    :param pboxes: coordinate of predicted boxes (array)
    :param scaling: boolean variable which allows to scale images (bool)
    :param plot_predictions: boolean for presence of target (bool)
    :param pathout: directory path for the plot image (Path)
    :return: None
    """

    if scaling:
        image_size = images[0].shape[0]
    else:
        image_size = 1
    plt.figure(figsize=(10, 10))
    for i in range(25):
        plt.subplot(5, 5, i + 1)
        plt.xticks([])
        plt.yticks([])
        plt.grid(False)
        plt.imshow(images[i], cmap="binary")
        center = (boxes[i, 0] * image_size, boxes[i, 1] * image_size)
        plt.gca().add_patch(
            Rectangle(
                center,
                (boxes[i, 2] - boxes[i, 0]) * image_size,
                (boxes[i, 3] - boxes[i, 1]) * image_size,
                edgecolor="red",
                facecolor="none",
                lw=4,
                label="observed",
            )
        )
        if plot_predictions:
            if pclasses[i][0] == 1:
                center = (pboxes[i, 0] * image_size, pboxes[i, 1] * image_size)
                plt.gca().add_patch(
                    Rectangle(
                        center,
                        (pboxes[i, 2] - pboxes[i, 0]) * image_size,
                        (pboxes[i, 3] - pboxes[i, 1]) * image_size,
                        edgecolor="blue",
                        facecolor="none",
                        lw=4,
                        label="predicted",
                    )
                )
            else:
                plt.gca().add_patch(
                    Rectangle(
                        (-1, -1),
                        0,
                        0,
                        edgecolor="blue",
                        facecolor="none",
                        lw=4,
                        label="predicted",
                    )
                )
        # plt.legend(loc="upper left", prop={"size": 6})
        plt.text(
            0.05 * image_size,
            0.14 * image_size,
            int(classes[i][0]),
            bbox=dict(facecolor="red", alpha=0.2),
        )
        plt.text(
            0.22 * image_size,
            0.14 * image_size,
            int(pclasses[i][0]),
            bbox=dict(facecolor="blue", alpha=0.2),
        )
    plt.legend(labels=["observed", "predicted"], bbox_to_anchor=(-1.33, -0.05))
    if save:
        pathout.mkdir(parents=True, exist_ok=True)
        plt.savefig(pathout / "validation_images_boxes.png")
        plt.close()
    else:
        plt.show()




[docs]def my_roc_plot(pathout, pred_train, obs_train, pred_val=None, obs_val=None, n_samples=1000, save=True):
    """
    ROC Curve plot.

    :param pathout: directory path for the plot image (Path)
    :param pred_train: predicted labels from training (int array)
    :param obs_train: labels of training images (int array)
    :param pred_val: predicted labels from validation (int array)
    :param obs_val: labels of validation images (int array)
    :param n_samples: number of samples (int)
    :return: rocT.opd["youden"].opt: ROC value obtained for training (double)
    :return: rocV.opd["youden"].opt: ROC value obtained for validation (double)
    """

    pos_label = True
    rocT = ru.compute_roc(
        X=pred_train,
        y=obs_train,
        pos_label=pos_label,
        objective=["minopt", "minoptsym", "youden", "cost", "concordance", "acc", "cohen"],
    )
    if all(pred_val) != None:
        rocV = ru.compute_roc(
            X=pred_val,
            y=obs_val,
            pos_label=pos_label,
            objective=["minopt", "minoptsym", "youden", "cost", "concordance", "acc", "cohen"],
        )
    # To perform a ROC analysis using bootstrapping
    plt.figure(figsize=(9, 7), dpi=80)
    plt.scatter(
        rocT["opd"]["youden"]["opp"][0],
        rocT["opd"]["youden"]["opp"][1],
        color="crimson",
        label="Opt point Train, tr: " + str(rocT.opd["youden"].opt),
    )
    ru.plot_roc_bootstrap(
        X=pred_train,
        y=obs_train,
        pos_label=pos_label,
        n_bootstrap=n_samples,
        color="crimson",
        label="Data Train",
    )
    if all(pred_val) != None:
        plt.scatter(
            rocV["opd"]["youden"]["opp"][0],
            rocV["opd"]["youden"]["opp"][1],
            color="royalblue",
            label="Opt point Val, tr: " + str(rocV.opd["youden"].opt),
        )
        ru.plot_roc_bootstrap(
            X=pred_val,
            y=obs_val,
            pos_label=pos_label,
            n_bootstrap=n_samples,
            color="royalblue",
            label="Data Val",
        )
    if save:
        plt.savefig(os.path.join(pathout, "ROC_train_val.png"))
        plt.close()
    else:
        plt.show()

    for key, val in rocT.opd.items():
        print(
            "train: %-15s thr=% .3f, J=%7.3f, opt point=%7.3f, %7.3f"
            % (key + ":", val.opt, val.opo, val.opp[0], val.opp[1])
        )
    print("")
    if all(pred_val) != None:
        for key, val in rocV.opd.items():
            print(
                "val: %-15s thr=% .3f, J=%7.3f, opt point=%7.3f, %7.3f"
                % (key + ":", val.opt, val.opo, val.opp[0], val.opp[1])
            )

    return rocT.opd["youden"].opt, rocV.opd["youden"].opt

[docs]def rtv_configuration_file(conf_file, path=None, save=True, pathout=None):
    """
    Save a copy of.yaml file in .txt format in pathout.

    :param conf_file: name of the configuration file (str)
    :param path: path of configuration file (Path)
    :param save: boolean variable to enable saving (bool)
    :param pathout: path for a copy of .yaml file (Path)
    :return: config: python object of configuration variables (object)
    """

    if path is not None:
        config_path = Path(path)
    else:
        config_path = Path.cwd().parent / "conf" / conf_file
    with open(config_path) as file:
        config = yaml.load(file, Loader=yaml.FullLoader)
    if save:
        if not os.path.exists(pathout):
            os.makedirs(pathout)
        with open(config_path, "r") as source, open(pathout / "conf.txt", "w") as dest:
            dest.write(source.read())
    return config

    """
    Random list of patient and random list of slices derived in the list of patient considered and in a range of central slices.

    :param n_patient: number of patients considered (int).
    :param n_random: number of random numbers to be generated (int).
    :param seed: seed used a starting point to generate random numbers (int).

[docs]def plot_crop_rectangle_over_image(
    image: np.ndarray,
    crop_rectangle: my_obj.Rectangle,
    isocentre: my_obj.Point = None,
    rectangle_color: str = None,
    isocentre_color: str = None,
    axes: plt.axes = None,
    padding: int = 0,
    **kwargs):
    """
    Plot CT slice for one patient with on top first guess rectangle (in blue) and adjusted rectangle (in red) to be
    used for crop and isocentre derived as red point.

    :param image: CT image slice (np.ndarray, shape (x,y)).
    :param crop_rectangle: Rectangle used for crop (Rectangle).
    :param isocentre: isocentre x and y positions (Point).
    :param rectangle_color: color of rectangle superimposed (str).
    :param isocentre_color: color of isocentre superimposed (str).
    :param axes: pre given axis. Default=None.
    :param padding: value used to enlarge the image to have the rectangle subscripted in it (int).
    :param kwargs: extra arguments.
    :return: axes: set of images.
    :return: im: output of axes.imshow().
    """

    if axes is None:
        fig, axes = plt.subplots(figsize=(6, 6))

    padded_image = np.pad(image, ((padding, padding), (padding, padding)), 'constant', constant_values=2000)
    im = axes.imshow(padded_image, cmap="Greys_r", vmin=-1000, vmax=1000)
    crop_rectangle.upper_left_x += padding
    crop_rectangle.upper_left_y += padding
    isocentre += my_obj.Point(padding, padding)
    if rectangle_color is not None:
        rectangle_plot_options = {
            "color": rectangle_color,
        }
    else:
        rectangle_plot_options = {}
    axes.plot(*crop_rectangle.draw(), **rectangle_plot_options, **kwargs)
    if isocentre is not None:
        if isocentre_color is not None:
            isocentre_plot_options = {
                "color": isocentre_color,
            }
        else:
            isocentre_plot_options = {}
        axes.scatter(isocentre.x, isocentre.y, **isocentre_plot_options, **kwargs)
    return axes, im

            i_slice = index_slice[i * 3 + j]
            single_image = image[patient_id].array_ct[i_slice, :, :]
            rectangle = my_obj.Rectangle(lower_right_point=image[patient_id].isocentre, width=crop_size)
            crop_rectangle = my_obj.CropRectangle(original_rectangle=rectangle, image_to_crop=single_image, epsilon=5)
            if plot_orig_rect:
                axes[i, j], im = plot_crop_rectangle_over_image(single_image, rectangle, image[patient_id].isocentre,
                                                                axes=axes[i, j], padding=padding)
            axes[i, j], im = plot_crop_rectangle_over_image(single_image, crop_rectangle, image[patient_id].isocentre,
                                                            axes=axes[i, j], color="red", padding=padding)
            axes[i, j].text(30 + padding, 512 - 50 + padding, f'ID: {image[patient_id].patient_id}',
                            bbox=dict(fill=True, edgecolor='white', facecolor='black', linewidth=1), fontsize=9,
                            color='w')

    [axi.set_axis_off() for axi in axes.ravel()]
    plt.tight_layout()
    # plt.subplots_adjust(wspace=0.01, hspace=0.01)
    fig.colorbar(im, ax=axes.ravel().tolist(), shrink=1., aspect=50, label="HU")  # , cax=cbaxes)
    if save:
        plt.savefig(str(pathout / f'multi_ct_crop_rects.png'), dpi=150)
        plt.close()
